Table User {
  id_user int [pk]
  nama varchar
  email varchar
  kata_sandi varchar
  ulang_kata_sandi varchar
  role enum
}

Table Anak {
  id_anak int [pk]
  id_user int [ref: > User.id_user]
  nama_anak varchar
  nik int
  tempat_lahir varchar
  tanggal_lahir date
  jenis_kelamin enum
  nama_ibu varchar
  nomer_hp int
  rukun_tangga int
  rukun_warga int
  kabupaten Enum
  kecamatan Enum
  desa enum
  alamat_jalan text
}

Table Ibu_Hamil {
  id_ibu int [pk]
  id_user int [ref: > User.id_user]
  nama_ibu varchar
  nik int
  tempat_lahir varchar
  tanggal_lahir date
  golongan_darah enum
  usia_kehamilan int
  nomer_hp int
  rukun_tangga int
  rukun_warga int
  kabupaten Enum
  kecamatan Enum
  desa enum
  alamat_jalan text
}

Table Penimbangan_Anak {
  id_penimbangan int [pk]
  tanggal_penimbangan date
  nama_anak enum [ref: > Anak.id_anak]
  umur_anak int
  berat_badan int
  tinggi_badan int
  lingkar_kepala int
  lingkar_lengan int
  petugas enum
  keterangan text
}

Table Imunisasi_Anak {
  id_imunisasi int [pk]
  tanggal_imunisasi date
  nama_anak enum [ref: > Anak.id_anak]
  jenis_vaksin enum
  petugas enum
  keterangan text
}

Table Vitamin_Anak {
  id_vitamin int [pk]
  tanggal_imunisasi date
  nama_anak enum [ref: > Anak.id_anak]
  jenis_vitamin enum
  dosis int
  petugas enum
  keterangan text
}

Table Pemeriksaan_Ibu_Hamil {
  id_pemeriksaan int [pk]
  tanggal_pemeriksaan date
  nama_ibu enum [ref: > Ibu_Hamil.id_ibu]
  tekanan_darah int
  berat_badan int
  lingkar_lengan int
  status_kehamilan enum
  petugas enum
  keterangan text
}

Table Laporan {
  id_laporan int [pk]
  id_user int [ref: > User.id_user]
  jenis_laporan enum
  total_value int
  jumlah_data int
}

Table Notifikasi_WhatsApp {
  id_notifikasi int [pk]
  id_user int [ref: > User.id_user]
  nomor_wa varchar
  jenis_notifikasi enum 
  pesan text
  waktu_kirim datetime
  status_pengiriman enum 
}

Table log_aktivitas (
  id_log INT AUTO_INCREMENT PRIMARY KEY,
  id_user INT NOT NULL,
  aktivitas TEXT NOT NULL,
  waktu DATETIME NOT NULL,
  FOREIGN KEY (id_user) REFERENCES users(id_user)
);
